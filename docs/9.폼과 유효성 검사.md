## 제어 컴포넌트와 비제어 컴포넌트

- 제어 컴포넌트: 상태를 리액트 컴포넌트가 제어하는 폼 요소. 입력값이 컴포넌트의 상태에 의해 제어됩니다.

**예제:**

```jsx
// 제어 컴포넌트
import React, { useState } from "react";

function ControlledForm() {
  const [value, setValue] = useState("");

  const handleChange = (event) => {
    setValue(event.target.value);
  };

  const handleSubmit = (event) => {
    event.preventDefault();
    alert("A name was submitted: " + value);
  };

  return (
    <form onSubmit={handleSubmit}>
      <label>
        Name:
        <input type="text" value={value} onChange={handleChange} />
      </label>
      <input type="submit" value="Submit" />
    </form>
  );
}

export default ControlledForm;
```

- 비제어 컴포넌트: DOM 요소가 직접 자신의 상태를 관리하는 폼 요소. 입력값이 DOM에서 직접 관리됩니다.

```jsx
// 비제어 컴포넌트
import React, { useRef } from "react";

function UncontrolledForm() {
  const inputRef = useRef(null);

  const handleSubmit = (event) => {
    event.preventDefault();
    alert("A name was submitted: " + inputRef.current.value);
  };

  return (
    <form onSubmit={handleSubmit}>
      <label>
        Name:
        <input type="text" ref={inputRef} />
      </label>
      <input type="submit" value="Submit" />
    </form>
  );
}

export default UncontrolledForm;
```

## 폼 라이브러리

폼 상태 관리 및 유효성 검사를 쉽게 처리할 수 있게 도와주는 라이브러리. 대표적으로 Formik과 React Hook Form이 있습니다.

**예제:**

```jsx
// Formik 예제
import React from "react";
import { Formik, Form, Field, ErrorMessage } from "formik";

function FormikExample() {
  return (
    <div>
      <h1>Sign Up</h1>
      <Formik
        initialValues={{ email: "", password: "" }}
        validate={(values) => {
          const errors = {};
          if (!values.email) {
            errors.email = "Required";
          } else if (
            !/^[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}$/i.test(values.email)
          ) {
            errors.email = "Invalid email address";
          }
          return errors;
        }}
        onSubmit={(values, { setSubmitting }) => {
          setTimeout(() => {
            alert(JSON.stringify(values, null, 2));
            setSubmitting(false);
          }, 400);
        }}
      >
        {({ isSubmitting }) => (
          <Form>
            <Field type="email" name="email" />
            <ErrorMessage name="email" component="div" />
            <Field type="password" name="password" />
            <ErrorMessage name="password" component="div" />
            <button type="submit" disabled={isSubmitting}>
              Submit
            </button>
          </Form>
        )}
      </Formik>
    </div>
  );
}

export default FormikExample;
```

```jsx
// React Hook Form 예제
import React from "react";
import { useForm } from "react-hook-form";

function ReactHookFormExample() {
  const {
    register,
    handleSubmit,
    watch,
    formState: { errors },
  } = useForm();
  const onSubmit = (data) => alert(JSON.stringify(data, null, 2));

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <label>Email</label>
      <input
        {...register("email", {
          required: true,
          pattern: /^[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}$/i,
        })}
      />
      {errors.email && <span>This field is required</span>}

      <label>Password</label>
      <input {...register("password", { required: true })} />
      {errors.password && <span>This field is required</span>}

      <input type="submit" />
    </form>
  );
}

export default ReactHookFormExample;
```

## 유효성 검사 라이브러리

폼 데이터의 유효성을 검사하는 라이브러리입니다. 대표적으로 Yup이 있으며, Formik과 같은 라이브러리와 함께 사용됩니다.

**예제:**

```jsx
// Formik과 Yup을 함께 사용하는 예제
import React from "react";
import { Formik, Form, Field, ErrorMessage } from "formik";
import * as Yup from "yup";

const SignupSchema = Yup.object().shape({
  email: Yup.string().email("Invalid email").required("Required"),
  password: Yup.string().min(6, "Password too short").required("Required"),
});

function FormikYupExample() {
  return (
    <div>
      <h1>Sign Up</h1>
      <Formik
        initialValues={{ email: "", password: "" }}
        validationSchema={SignupSchema}
        onSubmit={(values, { setSubmitting }) => {
          setTimeout(() => {
            alert(JSON.stringify(values, null, 2));
            setSubmitting(false);
          }, 400);
        }}
      >
        {({ isSubmitting }) => (
          <Form>
            <Field type="email" name="email" />
            <ErrorMessage name="email" component="div" />
            <Field type="password" name="password" />
            <ErrorMessage name="password" component="div" />
            <button type="submit" disabled={isSubmitting}>
              Submit
            </button>
          </Form>
        )}
      </Formik>
    </div>
  );
}

export default FormikYupExample;
```
